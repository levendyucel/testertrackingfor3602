    <script>
        if ('serviceWorker' in navigator) {
            navigator.serviceWorker.register('sw.js');
        }

        let data = { sicil: '', dept: '', time: '', items: [] };

        function nextStep(step) {
            if(step === 2 && !document.getElementById('sicil').value) return alert("Sicil girin!");
            document.querySelectorAll('#app > div').forEach(div => div.classList.add('hidden'));
            document.getElementById('step-' + step).classList.remove('hidden');
            if(step === 2) data.sicil = document.getElementById('sicil').value;
        }

        function setDept(dept) {
            data.dept = dept;
            document.getElementById('selected-dept').innerText = "📍 " + dept + " REYONU";
            nextStep(3);
        }

        function addMissing() {
            const name = document.getElementById('parfum-ad').value;
            const reason = document.getElementById('sebep').value;
            if(!name) return alert("Parfüm ismi boş olamaz!");
            data.items.push({ name, reason });
            const itemDiv = document.createElement('div');
            itemDiv.className = "bg-gray-100 p-2 rounded text-sm flex justify-between border-l-4 " + (reason === 'MERMA' ? 'border-red-500' : 'border-blue-500');
            itemDiv.innerHTML = `<span><strong>${name}</strong></span> <span class="text-xs uppercase">${reason}</span>`;
            document.getElementById('missing-list').appendChild(itemDiv);
            document.getElementById('parfum-ad').value = '';
        }

        function finishProcess(status) {
            data.time = new Date().toLocaleString('tr-TR');
            // Log verisi burada oluşuyor
            console.log("SUNUCUYA GİDEN VERİ:", data);
            
            document.querySelectorAll('#app > div').forEach(div => div.classList.add('hidden'));
            document.getElementById('step-finish').classList.remove('hidden');
            document.getElementById('summary').innerText = `${data.sicil} sicilli personel, ${data.time} saatinde ${data.dept} reyonu kontrolünü tamamladı.`;
        }
    </script>
